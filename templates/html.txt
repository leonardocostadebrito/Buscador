    <script>
        function mostrarDetalhes(celula) {
            var detalhes = celula.parentElement.querySelector('td.hidden');
            if (detalhes) {
                detalhes.classList.toggle('hidden');
            }
        }

        function pesquisarProcesso() {
            var input = document.getElementById("pesquisaProcesso");
            var filtro = input.value.toUpperCase();
            var tabela = document.getElementById("tabelaProcessos");
            var linhas = tabela.getElementsByTagName("tr");

            for (var i = 1; i < linhas.length; i++) {
                var celula = linhas[i].getElementsByTagName("td")[0];
                if (celula) {
                    var txtValor = celula.textContent || celula.innerText;
                    if (txtValor.toUpperCase().indexOf(filtro) > -1) {
                        linhas[i].style.display = "";
                    } else {
                        linhas[i].style.display = "none";
                    }
                }
            }
        }

        function baixarArquivo() {
            alert("Arquivo baixado com sucesso! (Simulação)");
        }
    </script>
